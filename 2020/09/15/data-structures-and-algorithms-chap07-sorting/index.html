冒泡排序

冒泡排序算法的原理如下：

  • 比较相邻的元素。如果第一个比第二个大，就交换他们两个。
  • 对每一对相邻元素做同样的工作，从开始第一对到结尾的最后一对。在这一点，最后的元素应该会是最大的数。
  • 针对所有的元素重复以上的步骤，除了最后一个。
  • 持续每次对越来越少的元素重复上面的步骤，直到没有任何一对数字需要比较。

时间复杂度

  • 若文件的初始状态是正序的，一趟扫描即可完成排序。所需的关键字比较次数 和记录移动次数 均达到最小值
    所以，冒泡排序最好的时间复杂度为 O(N)
  • 反序 O（N^2)

    public  static <Anytype extends Comparable<? super Anytype>>  void insertSort(Anytype[] arr){

        int j;
        for (int i = 1; i < arr.length; i++) {
            Anytype tmp=arr[i];

            for (j = 0; j < i; j++) {
                if(arr[j].compareTo(tmp)>0)
                    break;
            }


            for (int k=i; k >j; k--) {
                arr[k]=arr[k-1];
            }

            arr[j]=tmp;
        }

    }

    public  static <Anytype extends Comparable<? super Anytype>>  void insertSort1(Anytype[] arr) {

        for (int i = 1; i < arr.length; i++) {

                arr[j] = arr[j - 1];
            }
            arr[j] = tmp;
        }
    }

/**
 * @ClassName MyBinaryHeap

public class MyBinaryHeap<AnyType extends Comparable<? super AnyType>> {
    private static final int DEFAULT_SIZE=16;

    private int currentSize=0;
    private AnyType[] arr;

    MyBinaryHeap(){
        this(DEFAULT_SIZE);
    }

    MyBinaryHeap(int size){
        arr=(AnyType []) new Comparable[size+1];
    }

    public MyBinaryHeap(AnyType[] newArr){
        buildHeap(newArr);
    }

    public void insert(AnyType x){
        if(++currentSize>=arr.length){
            enlargeArray(arr.length*2+1);
        }

        int child=currentSize;
        //arr[child]=x;
        for (arr[0]=x;x.compareTo(arr[child/2])<0 ;child=child/2 ) {
            arr[child]=arr[child/2];
        }
        arr[child]=x;

    }



    public AnyType deleteMin(){

        if( isEmpty( ) )
            throw new UnderflowException( );


        AnyType x=findMin();
        arr[1]=arr[currentSize--];
        //arr[0]=arr[1];
        procolateDown(1);

        return x;
    }

    public boolean isEmpty() {
        return currentSize==0;
    }

    public void makeEmpty(){
        currentSize=0;
    }
    private void procolateDown(int i) {

        AnyType tmp=arr[i];
        int child;
        int hole;
        for ( hole = i;hole*2<=currentSize ;hole=child) {
            child=hole*2;

            if(arr[child+1]!=null&&arr[child+1].compareTo(arr[child])<0){
                child++;
            }

            if(tmp.compareTo(arr[child])>0){
                arr[hole]=arr[child];
            }
            else
                break;
        }
        arr[hole]=tmp;


    }


    private void buildHeap(AnyType[] newArr){

        int high=getHighBit(newArr.length);
        arr=(AnyType[]) new Comparable[1<<high+1];



        for (int i = 1; i < newArr.length+1; i++) {
            arr[i]=newArr[i-1];
        }
        currentSize=newArr.length;

        for (int i = currentSize/2; i>0; i--) {
            procolateDown(i);
        }
    }

    private int getHighBit(int length) {
        int high=1;
        while (length>>high!=0)
            high++;

        return high;
    }

    private AnyType findMin() {
        return arr[1];
    }

    private void enlargeArray(int size) {
        AnyType[] oldArr=arr;
        arr=(AnyType []) new Comparable[size];

        for (int i = 0; i <oldArr.length ; i++) {
            arr[i]=oldArr[i];
        }
    }

    public static void main( String [ ] args )
    {
        int numItems = 10000;
        MyBinaryHeap<Integer> h = new MyBinaryHeap<>( );
        int i = 37;

        for( i = 37; i != 0; i = ( i + 37 ) % numItems )
            h.insert( i );
        for( i = 1; i < numItems; i++ )
            if( h.deleteMin( ) != i )
                System.out.println( "Oops! " + i );


        Integer newArr[]=new Integer[1024];
        for (int j = 0; j <newArr.length ; j++) {
            newArr[j]=1024-j;
        }
        MyBinaryHeap<Integer> h1 = new MyBinaryHeap<>(newArr);
        System.out.println();

    }
}

    private static <AnyType extends Comparable<? super AnyType>>  void mergeSort(AnyType[]tmpArr,AnyType[] arr,int left,int right) {
        if(left<right){
            int center=(right+left)/2;
            mergeSort(tmpArr,arr,left,center);
            mergeSort(tmpArr,arr,center+1,right);
            merge(tmpArr,arr,left,(right+left)/2,right);

        }
    }

    private static  <AnyType extends Comparable<? super AnyType>>  void merge(AnyType[]tmpArr,AnyType[] arr, int left,int center, int right) {
        int leftBeign=left;

        int rightBeign=center+1;
        int rightEnd=right;

        int tmpPos=left;

        while(left<=center&&rightBeign<=right){

            if(arr[left].compareTo(arr[rightBeign])<0)
                tmpArr[tmpPos++]=arr[left++];

            else
                tmpArr[tmpPos++]=arr[rightBeign++];

        }

        while (left<=center)
            tmpArr[tmpPos++]=arr[left++];

        while (rightBeign<=right)
            tmpArr[tmpPos++]=arr[rightBeign++];

        for (int i = leftBeign; i <=rightEnd; i++) {
            arr[i]=tmpArr[i];
        }
    }

/**
 * @ClassName MyQuickSort
 * @Author: Wibur
 * @Description:

 * @Versuin: 1.0
 */
public class MyQuickSort {
    private static int CUT_OFF=10;
    public  static <Anytype extends Comparable<? super Anytype>>  void quickSort(Anytype[] arr) {
        quickSort(arr,0,arr.length-1);
    }

    private static <Anytype extends Comparable<? super Anytype>> void quickSort(Anytype[] arr, int left, int right) {

        if(right-left>CUT_OFF){
            Anytype prviot=Medium3(arr,left,right);

            int i=left,j=right-1;
            while(true){

                //注意在先++在运算 防止 arr[i]==a[j] 无限循环
                while (arr[++i].compareTo(prviot)<0);

                while (arr[--j].compareTo(prviot)>0);


                if(i<j)
                    swap(arr,i,j);
                else
                    break;
            }
            swap(arr,i,right-1);
            quickSort(arr,left,i-1);
            quickSort(arr,i+1,right);
        }

        else
            MyInsertSort.insertSort1(arr,left,right);
    }

    private static <Anytype extends Comparable<? super Anytype>> Anytype Medium3(Anytype[] arr, int left, int right) {

        int center=(left+right)/2;

        if(arr[center].compareTo(arr[left])<0)
            swap(arr,left,center);
        if(arr[right].compareTo(arr[left])<0)
            swap(arr,left,right);
        if(arr[right].compareTo(arr[center])<0)
            swap(arr,center,right);

        swap(arr,center,right-1);

        return arr[right-1];
    }

    private static <Anytype extends Comparable<? super Anytype>> void swap(Anytype[] arr, int indexX, int indexY) {
        Anytype tmp=arr[indexX];
        arr[indexX]=arr[indexY];
        arr[indexY]=tmp;
    }

package chap07;

import java.util.Arrays;

import java.util.stream.Stream;

/**
 * @ClassName MyQuickSort
 * @Author: Wibur
 * @Description:
 * @Date: Created in 14:50 2020/9/20
 * @Versuin: 1.0
 */
public class MyQuickSelect {
    private static int CUT_OFF=10;
    public  static <Anytype extends Comparable<? super Anytype>>  Anytype quickSelect(Anytype[] arr,int k) {
        return quickSelect(arr,0,arr.length-1,k);
    }

    private static <Anytype extends Comparable<? super Anytype>> Anytype quickSelect(Anytype[] arr, int left, int right,int k) {

        if(right-left>CUT_OFF){
            Anytype prviot=Medium3(arr,left,right);

            int i=left,j=right-1;
            while(true){

                //注意在先++在运算 防止 arr[i]==a[j] 无线循环
                while (arr[++i].compareTo(prviot)<0);

                while (arr[--j].compareTo(prviot)>0);


                if(i<j)
                    swap(arr,i,j);
                else
                    break;
            }
            swap(arr,i,right-1);

            int diff=i-left+1;
            if(diff>k) {
                return quickSelect(arr, left, i - 1, k);
            }
            else if(diff<k) {
                return quickSelect(arr, i + 1, right,k-diff);
            }
            else {
                return arr[i];
            }
        }

        else{
            MyInsertSort.insertSort1(arr,left,right);
            return arr[left+k-1];
        }
    }

    private static <Anytype extends Comparable<? super Anytype>> Anytype Medium3(Anytype[] arr, int left, int right) {

        int center=(left+right)/2;

        if(arr[center].compareTo(arr[left])<0)
            swap(arr,left,center);
        if(arr[right].compareTo(arr[left])<0)
            swap(arr,left,right);
        if(arr[right].compareTo(arr[center])<0)
            swap(arr,center,right);

        swap(arr,center,right-1);

        return arr[right-1];
    }

    private static <Anytype extends Comparable<? super Anytype>> void swap(Anytype[] arr, int indexX, int indexY) {
        Anytype tmp=arr[indexX];
        arr[indexX]=arr[indexY];
        arr[indexY]=tmp;
    }

    public static void main(String[] args) {

        Random r=new Random();
        r.setSeed(0);
        int[] nums=new int[1000];
        int upBound=1000;

        for (int i = 0; i < nums.length; i++) {
            nums[i]=r.nextInt(upBound);
        }
        //int[] nums = {3,1,4,1,5,9,2,6,5,6,9,10};
//先将int数组转换为数值流
        IntStream stream = Arrays.stream(nums);
//流中的元素全部装箱，转换为流 ---->int转为Integer
        Stream<Integer> integerStream = stream.boxed();
//将流转换为数组
        Integer[] integers = integerStream.toArray(Integer[]::new);
        MyQuickSort.quickSort(integers);
        System.out.println(integers[50-1]);
        System.out.println(quickSelect(integers,50));


    }
}

package chap07;

import java.util.ArrayList;
import java.util.List;
import java.util.Arrays;
import java.util.Random;

public class RadixSort
{
        
    /*
     * Radix sort an array of Strings
     * Assume all are all ASCII
     * Assume all have length bounded by maxLen
     */
     //方法1 适用于不同length 位置从后往前
    public static void radixSort( String [ ] arr, int maxLen )
    {
        final int BUCKETS = 256;
        
        ArrayList<String> []wordsByLength = new ArrayList[ maxLen + 1 ];
        ArrayList<String> []buckets = new ArrayList[ BUCKETS ];
        
        for( int i = 0; i < wordsByLength.length; i++ )
            wordsByLength[ i ] = new ArrayList<>( );
        
        for( int i = 0; i < BUCKETS; i++ )
            buckets[ i ] = new ArrayList<>( );
        
        for( String s : arr )
            wordsByLength[ s.length( ) ].add( s );
       
        int idx = 0;
        for( ArrayList<String> wordList : wordsByLength )
            for( String s : wordList )
                arr[ idx++ ] = s;
        
        int startingIndex = arr.length;    
        for( int pos = maxLen - 1; pos >= 0; pos-- )
        {
            startingIndex -= wordsByLength[ pos + 1 ].size( );
            
            for( int i = startingIndex; i < arr.length; i++ )
                buckets[ arr[ i ].charAt( pos ) ].add( arr[ i ] );
            
            idx = startingIndex;
            for( ArrayList<String> thisBucket : buckets )
            {
                for( String s : thisBucket )
                    arr[ idx++ ] = s;
                
                thisBucket.clear( );
            }
        }
    }
    /*
     * Radix sort an array of Strings
     * Assume all are all ASCII
     * Assume all have same length
     */
    public static void radixSortA( String [ ] arr, int stringLen )
    {
        final int BUCKETS = 256;

        ArrayList<String>[] buckets = new ArrayList[BUCKETS];

        for( int i = 0; i < BUCKETS; i++ )
            buckets[ i ] = new ArrayList<>( );
        
        for( int pos = stringLen - 1; pos >= 0; pos-- )
        {
            for( String s : arr )
                buckets[ s.charAt( pos ) ].add( s );
            
            int idx = 0;
            for( ArrayList<String> thisBucket : buckets )
            {
                for( String s : thisBucket )
                    arr[ idx++ ] = s;
                
                thisBucket.clear( );
            }
        }
    }
       
    /*
     * Counting radix sort an array of Strings
     * Assume all are all ASCII
     * Assume all have same length
     */
    public static void countingRadixSort( String [ ] arr, int stringLen )
    {
        final int BUCKETS = 256;
        
        int N = arr.length;
        String [ ] buffer = new String[ N ];

        String [ ] in = arr;
        String [ ] out = buffer;
        
        for( int pos = stringLen - 1; pos >= 0; pos-- )
        {
            int[ ] count = new int [ BUCKETS + 1 ];
            
            for( int i = 0; i < N; i++ )
                count[ in[ i ].charAt( pos ) + 1 ]++;

            for( int b = 1; b <= BUCKETS; b++ )
                count[ b ] += count[ b - 1 ];

            for( int i = 0; i < N; i++ )
                out[ count[ in[ i ].charAt( pos ) ]++ ] = in[ i ];
            
              // swap in and out roles
            String [ ] tmp = in;
            in = out;
            out = tmp;
        }
        
           // if odd number of passes, in is buffer, out is arr; so copy back
        if( stringLen % 2 == 1 )
            for( int i = 0; i < arr.length; i++ )
                out[ i ] = in[ i ];
    }
